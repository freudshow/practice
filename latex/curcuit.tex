    \begin{circuitikz}

      to[V,v=$U_q$] (0,2) % 电压源
      to[short] (2,2)%坐标(2,2)做为起始点，(2,0)做为终点，绘制电阻。_R_代表电压源，_R=$R_1$_绘制标识。
      to[R=$R_1$] (2,0) % 电阻
      to[short] (0,0);%注意结尾的分号！
    \end{circuitikz}
